648 $parser = Log::Log4perl::Config::DOMConfigurator->new();
649 $data = $parser->parse(\@text);
650 } else {
651 $parser = Log::Log4perl::Config::PropertyConfigurator->new();
652 $data = $parser->parse(\@text);
653 }

688############################################################
689sub leaf_paths {
690############################################################
691# Takes a reference to a hash of hashes structure of
692# arbitrary depth, walks the tree and returns a reference
693# to an array of all possible leaf paths (each path is an
694# array again).
695# Example: { a => { b => { c => d }, e => f } } would generate
696# [ [a, b, c, d], [a, e, f] ]
697############################################################
698 my ($root) = @_;
699
700 my @stack = ();
701 my @result = ();
702
703 push @stack, [$root, []];
704
705 while(@stack) {
706 my $item = pop @stack;
707
708 my($node, $path) = @$item;
709
710 if(ref($node) eq "HASH") {
711 for(keys %$node) {
712 push @stack, [$node->{$_}, [@$path, $_]];
713 }
714 } else {
715 push @result, [@$path, $node];
716 }
717 }
718 return \@result;
719}

748sub compile_if_perl {
749###########################################
750 my($value) = @_;
751
752 if($value =~ /^\s*sub\s*{/ ) {
753 my $mask;
754 unless( Log::Log4perl::Config->allow_code() ) {
755 die "\$Log::Log4perl::Config->allow_code() setting " .
756 "prohibits Perl code in config file";
757 }
758 if( defined( $mask = Log::Log4perl::Config->allowed_code_ops() ) ) {
759 return compile_in_safe_cpt($value, $mask );
760 }
761 elsif( $mask = Log::Log4perl::Config->allowed_code_ops_convenience_map(
762 Log::Log4perl::Config->allow_code()
763 ) ) {
764 return compile_in_safe_cpt($value, $mask );
765 }
766 elsif( Log::Log4perl::Config->allow_code() == 1 ) {
767
768 # eval without restriction
769 my $cref = eval "package main; $value" or
770 die "Can't evaluate '$value' ($@)";
771 return $cref;
772 }
773 else {
774 die "Invalid value for \$Log::Log4perl::Config->allow_code(): '".
775 Log::Log4perl::Config->allow_code() . "'";
776 }
777 }
778
779 return undef;
780}

783sub compile_in_safe_cpt {
784###########################################
785 my($value, $allowed_ops) = @_;
786
787 # set up a Safe compartment
788 require Safe;
789 my $safe = Safe->new();
790 $safe->permit_only( @{ $allowed_ops } );
791
792 # share things with the compartment
793 for( keys %{ Log::Log4perl::Config->vars_shared_with_safe_compartment() } ) {
794 my $toshare = Log::Log4perl::Config->vars_shared_with_safe_compartment($_);
795 $safe->share_from( $_, $toshare )
796 or die "Can't share @{ $toshare } with Safe compartment";
797 }
798
799 # evaluate with restrictions
800 my $cref = $safe->reval("package main; $value") or
801 die "Can't evaluate '$value' in Safe compartment ($@)";
802 return $cref;
803
804}

820###########################################

952=head1 NAME
953
954Log::Log4perl::Config - Log4perl configuration file syntax
955
956=head1 DESCRIPTION
957
958In C<Log::Log4perl>, configuration files are used to describe how the
959system's loggers ought to behave.
960
961The format is the same as the one as used for C<log4j>, just with
962a few perl-specific extensions, like enabling the C<Bar::Twix>
963syntax instead of insisting on the Java-specific C<Bar.Twix>.
964
965Comment lines (starting with arbitrary whitespace and a #) and
966blank lines (all whitespace or empty) are ignored.
967
968Also, blanks between syntactical entities are ignored, it doesn't
969matter if you write
970
971 log4perl.logger.Bar.Twix=WARN,Screen
972
973or
974
975 log4perl.logger.Bar.Twix = WARN, Screen
976
977C<Log::Log4perl> will strip the blanks while parsing your input.
978
979Assignments need to be on a single line. However, you can break the
980line if you want to by using a continuation character at the end of the
981line. Instead of writing
982
983 log4perl.appender.A1.layout=Log::Log4perl::Layout::SimpleLayout
984
985you can break the line at any point by putting a backslash at the very (!)
986end of the line to be continued:
987
988 log4perl.appender.A1.layout=\
989 Log::Log4perl::Layout::SimpleLayout
990
991Watch out for trailing blanks after the backslash, which would prevent
992the line from being properly concatenated.
993
994=head2 Loggers
995
996Loggers are addressed by category:
997
998 log4perl.logger.Bar.Twix = WARN, Screen
999
1000This sets all loggers under the C<Bar::Twix> hierarchy on priority
1001C<WARN> and attaches a later-to-be-defined C<Screen> appender to them.
1002Settings for the root appender (which doesn't have a name) can be
1003accomplished by simply omitting the name:
1004
1005 log4perl.logger = FATAL, Database, Mailer
1006
1007This sets the root appender's level to C<FATAL> and also attaches the
1008later-to-be-defined appenders C<Database> and C<Mailer> to it.
1009
1010The additivity flag of a logger is set or cleared via the
1011C<additivity> keyword:
1012
1013 log4perl.additivity.Bar.Twix = 0|1
1014
1015(Note the reversed order of keyword and logger name, resulting
1016from the dilemma that a logger name could end in C<.additivity>
1017according to the log4j documentation).
1018
1019=head2 Appenders and Layouts
1020
1021Appender names used in Log4perl configuration file
1022lines need to be resolved later on, in order to
1023define the appender's properties and its layout. To specify properties
1024of an appender, just use the C<appender> keyword after the
1025C<log4perl> intro and the appender's name:
1026
1027 # The Bar::Twix logger and its appender
1028 log4perl.logger.Bar.Twix = DEBUG, A1
1029 log4perl.appender.A1=Log::Log4perl::Appender::File
1030 log4perl.appender.A1.filename=test.log
1031 log4perl.appender.A1.mode=append
1032 log4perl.appender.A1.layout=Log::Log4perl::Layout::SimpleLayout
1033
1034This sets a priority of C<DEBUG> for loggers in the C<Bar::Twix>
1035hierarchy and assigns the C<A1> appender to it, which is later on
1036resolved to be an appender of type C<Log::Log4perl::Appender::File>, simply
1037appending to a log file. According to the C<Log::Log4perl::Appender::File>
1038manpage, the C<filename> parameter specifies the name of the log file
1039and the C<mode> parameter can be set to C<append> or C<write> (the
1040former will append to the logfile if one with the specified name
1041already exists while the latter would clobber and overwrite it).
1042
1043The order of the entries in the configuration file is not important,
1044C<Log::Log4perl> will read in the entire file first and try to make
1045sense of the lines after it knows the entire context.
1046
1047You can very well define all loggers first and then their appenders
1048(you could even define your appenders first and then your loggers,
1049but let's not go there):
1050
1051 log4perl.logger.Bar.Twix = DEBUG, A1
1052 log4perl.logger.Bar.Snickers = FATAL, A2
1053
1054 log4perl.appender.A1=Log::Log4perl::Appender::File
1055 log4perl.appender.A1.filename=test.log
1056 log4perl.appender.A1.mode=append
1057 log4perl.appender.A1.layout=Log::Log4perl::Layout::SimpleLayout
1058
1059 log4perl.appender.A2=Log::Log4perl::Appender::Screen
1060 log4perl.appender.A2.stderr=0
1061 log4perl.appender.A2.layout=Log::Log4perl::Layout::PatternLayout
1062 log4perl.appender.A2.layout.ConversionPattern = %d %m %n
1063
1064Note that you have to specify the full path to the layout class
1065and that C<ConversionPattern> is the keyword to specify the printf-style
1066formatting instructions.
1067
1068=head1 Configuration File Cookbook
1069
1070Here's some examples of often-used Log4perl configuration files:
1071
1072=head2 Append to STDERR
1073
1074 log4perl.category.Bar.Twix = WARN, Screen
1075 log4perl.appender.Screen = Log::Log4perl::Appender::Screen
1076 log4perl.appender.Screen.layout = \
1077 Log::Log4perl::Layout::PatternLayout
1078 log4perl.appender.Screen.layout.ConversionPattern = %d %m %n
1079
1080=head2 Append to STDOUT
1081
1082 log4perl.category.Bar.Twix = WARN, Screen
1083 log4perl.appender.Screen = Log::Log4perl::Appender::Screen
1084 log4perl.appender.Screen.stderr = 0
1085 log4perl.appender.Screen.layout = \
1086 Log::Log4perl::Layout::PatternLayout
1087 log4perl.appender.Screen.layout.ConversionPattern = %d %m %n
1088
1089=head2 Append to a log file
1090
1091 log4perl.logger.Bar.Twix = DEBUG, A1
1092 log4perl.appender.A1=Log::Log4perl::Appender::File
1093 log4perl.appender.A1.filename=test.log
1094 log4perl.appender.A1.mode=append
1095 log4perl.appender.A1.layout = \
1096 Log::Log4perl::Layout::PatternLayout
1097 log4perl.appender.A1.layout.ConversionPattern = %d %m %n
1098
1099Note that you could even leave out
1100
1101 log4perl.appender.A1.mode=append
1102
1103and still have the logger append to the logfile by default, although
1104the C<Log::Log4perl::Appender::File> module does exactly the opposite.
1105This is due to some nasty trickery C<Log::Log4perl> performs behind
1106the scenes to make sure that beginner's CGI applications don't clobber
1107the log file every time they're called.
1108
1109=head2 Write a log file from scratch
1110
1111If you loathe the Log::Log4perl's append-by-default strategy, you can
1112certainly override it:
1113
1114 log4perl.logger.Bar.Twix = DEBUG, A1
1115 log4perl.appender.A1=Log::Log4perl::Appender::File
1116 log4perl.appender.A1.filename=test.log
1117 log4perl.appender.A1.mode=write
1118 log4perl.appender.A1.layout=Log::Log4perl::Layout::SimpleLayout
1119
1120C<write> is the C<mode> that has C<Log::Log4perl::Appender::File>
1121explicitely clobber the log file if it exists.
